### AssemblyLoad示例
        class HPACK_Codec
        {
        public Assembly _dllAssembly;

        //可选
        private Type? _logType; 
        private FieldInfo? _infoField;
        private FieldInfo? _warningField;
        private FieldInfo? _errorField;

        private Type? _hpackEncoderType;
        private Type? _hpackDecoderType;
        public Type? _HpackDynamicTable;
        public Type? _IndexType;
        public MethodInfo? _encodeMethod;
        public MethodInfo? _decodeMethod;
       
        public HPACK_Codec(string dllPath)
        {
            _dllAssembly = Assembly.LoadFrom(dllPath);

            _hpackEncoderType = _dllAssembly.GetType("HPACK_Codec.HPACK_Encode");
            if (_hpackEncoderType == null)
                throw new TypeLoadException("HpackEncoder class not found in DLL");
            _hpackDecoderType = _dllAssembly.GetType("HPACK_Codec.HPACK_Decode");
            if (_hpackEncoderType == null)
                throw new TypeLoadException("HpackDecoder class not found in DLL");
            _HpackDynamicTable = _dllAssembly.GetType("HPACK_Codec.core.HpackDynamicTable");
            if (_HpackDynamicTable == null)
                throw new TypeLoadException("HpackDynamicTable class not found in DLL");
            _IndexType= _dllAssembly.GetType("HPACK_Codec.core.IndexType");
            if (_IndexType == null)
                throw new TypeLoadException("IndexType enum not found in DLL");
            _encodeMethod = _hpackEncoderType.GetMethod("Encode",BindingFlags.Public | BindingFlags.Static,null,
                new Type[] {typeof(string), typeof(string), _HpackDynamicTable,typeof(bool), _IndexType},null);
            if (_encodeMethod == null)
                throw new MissingMethodException("Encode method not found");
            _decodeMethod = _hpackDecoderType!.GetMethod("Decode",BindingFlags.Public | BindingFlags.Static,null,
                new Type[] {typeof(byte[]), _HpackDynamicTable,},null);
            if (_decodeMethod == null)
                throw new MissingMethodException("Encode method not found");
        }
        // 启用日志
        public void LogInit()
        {
            // 2. 获取Log类型（注意完整命名空间）
            _logType = _dllAssembly.GetType("HPACK_Codec.Log");
            if (_logType == null)
                throw new TypeLoadException("Log class not found in DLL");        
            // 3. 获取Info字段
            _infoField = _logType.GetField("Info", BindingFlags.Public | BindingFlags.Static);
            if (_infoField == null )
                throw new MissingFieldException("LogInformation field not found");
            _errorField = _logType.GetField("Error", BindingFlags.Public | BindingFlags.Static);
            if (_errorField == null )
                throw new MissingFieldException("LogError field not found");
            _warningField = _logType.GetField("Warning", BindingFlags.Public | BindingFlags.Static);
            if (_warningField == null )
                throw new MissingFieldException("LogWarning field not found");
            

        }
        public void InfoSubscribe(Action<string> logHandler)=> _infoField!.SetValue(null, Delegate.CreateDelegate(
                _infoField!.FieldType,
                logHandler.Target,
                logHandler.Method
            ));
        public void WarningSubscribe(Action<string> logHandler) => _warningField!.SetValue(null, Delegate.CreateDelegate(
                _warningField!.FieldType,
                logHandler.Target,
                logHandler.Method
            ));
        public void ErrorSubscribe(Action<string> logHandler) => _errorField!.SetValue(null, Delegate.CreateDelegate(
                _errorField!.FieldType,
                logHandler.Target,
                logHandler.Method
            ));

    }
